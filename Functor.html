Library Functor

Require Import Function.

Class Functor (functor: Type → Type) := {
    fmap: ∀ {A B : Type}, (A → B) → functor A → functor B;

    functors_preserve_composition
        : ∀ (A B C: Type)
        , ∀ (g : B → C)
        , ∀ (f : A → B)
        , ∀ (c : functor A)
        , fmap g (fmap f c) = fmap (compose g f) c;

    functor_id_law
        : ∀ (A : Type), ∀ (c : functor A), fmap id c = id c
}.
